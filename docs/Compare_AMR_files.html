# Clean Resfinder gene names
resfinder_genes = {gene.replace('.match', '') for gene in df_res.columns}
resfinder_genes2 = [w.replace('_', '') for w in resfinder_genes]
print (resfinder_genes2)
# Get PW AMR gene names
pw_amr_genes = set(df_pw .columns)

# Find substrings
substr_in_resfinder = {gene for gene in pw_amr_genes if any(gene in res_gene for res_gene in resfinder_genes2)}

print("PW AMR genes that are substrings of Resfinder genes:", substr_in_resfinder)
